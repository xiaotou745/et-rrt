USE RenRenDB
alter table dbo.TaskDatum add RefuReason NVARCHAR(200) NOT NULL default('') 
--添加列注释
EXEC sys.sp_addextendedproperty @name=N'MS_Description',
  @value=N'拒绝原因' ,
  @level0type=N'SCHEMA',
  @level0name=N'dbo', 
  @level1type=N'TABLE',
  @level1name=N'TaskDatum', 
  @level2type=N'COLUMN',
  @level2name=N'RefuReason'
GO
