--添加列注释

  @value=N'任务总价格(商家给平台的价格)' ,
  @level0type=N'SCHEMA',
  @level0name=N'dbo', 
  @level1type=N'TABLE',
  @level1name=N'RenRenTask', 
  @level2type=N'COLUMN',
  @level2name=N'TotalAmount'
GO
